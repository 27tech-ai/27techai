Tracking Setup

Add this snippet to your website's index.html to start tracking visitors and clicks.

<script>
const DASHBOARD_URL = window.location.hostname === 'localhost'
    ? 'http://localhost:5000'
    : 'http://YOUR_IP:5000';

// Track page views
fetch(DASHBOARD_URL + '/api/tracking', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
        event: 'page_view',
        page: window.location.pathname,
        referrer: document.referrer,
        user_agent: navigator.userAgent,
        device: /Mobi|Android/i.test(navigator.userAgent) ? 'mobile' : 'desktop'
    })
});

// Track script clicks
document.addEventListener('click', function(e) {
    const link = e.target.closest('a[href]');
    if (link) {
        const card = link.closest('.card-hover, [class*=card]');
        const title = card ? card.querySelector('h3') : null;
        if (title) {
            fetch(DASHBOARD_URL + '/api/tracking', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    event: 'script_click',
                    script_title: title.textContent,
                    referrer: document.referrer
                })
            });
        }
    }
});
</script>
